SELECT
`from_places`.`name` AS `Quelle`,

`potentials`.`miv-change` AS `MIV-Veränderung`,

LEFT JOIN `places` `from_places` ON `from_places`.`id` = `potentials`.`from_id`

WHERE `from_places`.`name` = "Mönchstockheim";
